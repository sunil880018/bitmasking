class Solution {
public:
    bool isPowerOfFour(int num) {
         if(num==1||num==4)
                  return 1;
         if(num<16)
                  return 0;
            while(num>3)
            {
                    if(num%4<=3&&num%4!=0) return 0;
                    num/=4;
            }
                    
            if(num==1)
               return 1;
            return 0;
    }
};
